// MÜZİK SİSTEMİ AYARLARI
const playlist = ["ozan.mp3", "eyvah.mp3", "catmayarim.mp3", "guldoktum.mp3", "bizegidelim.mp3", "buyumedim.mp3"];
let suankiSarkiIndex = 0;
let isPlaying = false;
const player = document.getElementById("anaMuzik");
const playBtn = document.getElementById("play-pause-btn");

function muzikYukle(index) {
    player.src = playlist[index];
    document.getElementById("sarki-bilgi").innerText = "Çalıyor: Şarkı " + (index + 1);
}

function muzikKontrol() {
    if (!player.src) muzikYukle(suankiSarkiIndex);
    if (isPlaying) {
        player.pause();
        playBtn.innerText = "▶";
        playBtn.classList.remove("playing");
    } else {
        player.play();
        playBtn.innerText = "⏸";
        playBtn.classList.add("playing");
    }
    isPlaying = !isPlaying;
}

function sonrakiSarki() {
    suankiSarkiIndex = (suankiSarkiIndex + 1) % playlist.length;
    muzikYukle(suankiSarkiIndex);
    player.play();
    isPlaying = true;
    playBtn.innerText = "⏸";
    playBtn.classList.add("playing");
}

function oncekiSarki() {
    suankiSarkiIndex = (suankiSarkiIndex - 1 + playlist.length) % playlist.length;
    muzikYukle(suankiSarkiIndex);
    player.play();
    isPlaying = true;
    playBtn.innerText = "⏸";
    playBtn.classList.add("playing");
}

function rastgeleSarki() {
    suankiSarkiIndex = Math.floor(Math.random() * playlist.length);
    muzikYukle(suankiSarkiIndex);
    player.play();
    isPlaying = true;
    playBtn.innerText = "⏸";
    playBtn.classList.add("playing");
}

        function saatiGuncelle() {
            const simdi = new Date();
            const s = String(simdi.getHours()).padStart(2, '0');
            const d = String(simdi.getMinutes()).padStart(2, '0');
            const sn = String(simdi.getSeconds()).padStart(2, '0');
            document.getElementById("saat-kutusu").innerText = `${s}:${d}:${sn}`;
        }
        setInterval(saatiGuncelle, 1000);
        saatiGuncelle();

        function ilkHazirlik() {
            secilenKelime = kelimeListesi[Math.floor(Math.random() * kelimeListesi.length)];

            document.getElementById("tahmin").focus();
        }

            oyunBasladi = true;
            genelTimer = setInterval(() => {
                oyunSuresi--;
                document.getElementById("genel-sure").innerText = oyunSuresi;
                if(oyunSuresi <= 10) document.getElementById("genel-sure").classList.add("panik");
                if(oyunSuresi <= 0) gameOver();
            }, 1000);
        }

        function yeniSoru() {
            if(oyunSuresi <= 0) return;
            secilenKelime = kelimeListesi[Math.floor(Math.random() * kelimeListesi.length)];
            document.getElementById("karisik-kelime").innerText = secilenKelime.toUpperCase();

            kelimeYazilmayaBaslandi = false;
        }

        function triggerHyperSpeed() {
            document.body.classList.add("light-speed-active");

        }

        function gameOver() {
            clearInterval(genelTimer);
            let isNew = false;
            if (toplamDogru > bestCorrect) { 
                localStorage.setItem("secoBestCorrect", toplamDogru); 
                bestCorrect = toplamDogru;
                isNew = true; 
            }
            document.getElementById("game-over").style.display = "flex";
            if(isNew) {
                document.getElementById("new-record-msg").style.display = "block";
                confetti({ particleCount: 150, spread: 70, origin: { y: 0.6 } });
            }
            let yuzde = Math.round((toplamDogru / (toplamDogru + toplamYanlis || 1)) * 100);
            document.getElementById("final-stats").innerHTML = `
                Toplam Doğru: ${toplamDogru}<br>
                Toplam Yanlış: ${toplamYanlis}<br>
                İsabet Oranı: %${yuzde}
            `;
        }

                bestCorrect = 0;
                document.getElementById("best-correct").innerText = "0";
                alert("Rekor sıfırlandı!");
            }
        }
